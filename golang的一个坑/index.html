Golang的一个坑

December 9, 2017

试着输入以下代码：

package main

import "fmt"
import "sync"

func main() {
	a := make([]int, 5)
	var wg sync.WaitGroup
	wg.Add(5)
	for i, j := range a {
		go func() {
			fmt.Println(i, j)
			wg.Done()
		}()
	}
	wg.Wait()
}
